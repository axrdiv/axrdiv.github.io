2. Tree Matching

树中有个节点，定义Matching为edge所组成的集合，所有的edge中不包含重复的节点。求最大Matching集合的大小。

这道题需要用到DP方法。从一个度为1的节点出发，考虑两种情况：1.该节点不在Matching中；2.该节点在Matching中，且为edge的第一个节点。因而有公式

int dp[2][maxn];

/*
 * 0 means won't select it,
 * 1 means select it as edge's first node;
 * dp[0][s] = sum(max(dp[0][u], dp[1][u]))
 * dp[1][s] = max(dp[0][u_1] + sum max(dp[0][u_2], dp[1][u_2])) + 1;
 */
void dfs(int s, int e) {
    dp[0][s] = 0;
    dp[1][s] = 0;
    for(auto u : adj[s]) {
        if(u == e) continue;
        dfs(u, s);
    }

    for(auto u : adj[s]) {
        if(u == e) continue;
        dp[0][s] += max(dp[0][u], dp[1][u]);
    }

    for(auto u : adj[s]) {
        if(u == e) continue;
        dp[1][s] = max(dp[1][s], dp[0][s] + (dp[0][u] > dp[1][u] ? 0 : dp[0][u] - dp[1][u]) + 1);
    }
}

3. Tree Diameter

树的直径是树中相距最远的两个节点的距离。简单的方法是使用两次DFS，首先以任意节点为根找出最远的叶子节点，之后以该叶子节点为根，再次使用DFS找出最远叶子节点。这两个叶子节点的距离即为树的直径。

int maxd;

void dfs(int s, int e, int d) {
    if(d > maxd) {
        maxd = d;
        id = s;
    }
    for(auto u : adj[s]) {
        if(u == e) continue;
        dfs(u, s, d+1);
    }
}
int main() {
    // input code
    dfs(1, 0, 0);
    int st = id;
    maxd = 0;
    dfs(st, 0, 0);
    // output code
}

4. Tree Distances I

计算每个节点到最远节点的距离。可以用Floyd-Warshall算法，但是复杂度肯定会超时。因而需要用到树上DP。

计算也是通过两次DFS。第一次DFS计算子节点方向上的最长距离，复杂度。第二次DFS计算父节点方向上的最长距离。具体方法见Blog。

int dp[3][maxn];

void dfs_downwards(int s, int e) {
    dp[0][s] = 0;
    for(auto u : adj[s]) {
        if(u == e) continue;
        dfs_downwards(u, s);
        dp[0][s] = max(dp[0][s], dp[0][u]+1);
    }
}

void dfs_upwards(int s, int e) {
    if(e && dp[1][e] == dp[0][s] + 1) dp[1][s] = dp[2][e] + 1;
    else if(e) dp[1][s] = dp[1][e] + 1;

    for(auto u : adj[s]) {
        if(u == e) continue;
        if(dp[1][s] < dp[0][u]+1) {
            swap(dp[1][s], dp[2][s]);
            dp[1][s] = dp[0][u]+1;
        }
        else if(dp[2][s] < dp[0][u]+1) {
            dp[2][s] = dp[0][u]+1;
        }
    }

    for(auto u : adj[s]) {
        if(u == e) continue;
        dfs_upwards(u, s);
    }
}

int main() {
    // input code
    dfs_upwards(1, 0);
    dfs_downwards(1, 0);
    // output code
}
